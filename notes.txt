Capture rsync output:

RSYNC_OUTPUT=$(rsync -irtS --no-motd --delete ...)
if [ $? -eq 0 ]; then
    if [ -n "${RSYNC_OUTPUT}" ]; then
        # There were changes
        exit 200
    else
        # No changes were made by rsync
        exit 0
    fi
else
    # Something went wrong!
    exit 1
fi
